#include <tunables/global>

profile autossh flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Allow broadly for file, signal and network so s6-overlay, bashio and the
  # SSH tunnel work without per-path maintenance. Pattern mirrors
  # home-assistant/addons/dnsmasq. Security boundary is shifted to the
  # explicit capability allowlist and the deny rules below.
  file,
  signal,
  network,

  # Capabilities: only what autossh + ssh actually need.
  capability net_bind_service,
  capability setgid,
  capability setuid,
  capability dac_read_search,
  capability chown,
  capability fowner,

  # Explicit escalation denies.
  deny /proc/*/oom_score_adj w,
  deny /sys/kernel/** w,
  deny /sys/fs/cgroup/** w,
  deny mount,
  deny umount,
  deny pivot_root,
  deny ptrace,
}
